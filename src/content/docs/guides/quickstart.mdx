Get from zero to processing background jobs in under a minute. This guide starts an OJS server with Docker, then walks you through enqueuing and processing a job in your language of choice.

:::tip[Looking for in-depth tutorials?]
After this quickstart, check out the language-specific [Your First Job tutorials](/tutorials/your-first-job-go/) for a deeper walkthrough.
:::

## Prerequisites

- [Docker](https://docs.docker.com/get-docker/) and Docker Compose installed
- One of: Go 1.24+, Node.js 18+, Python 3.11+, Ruby 3.2+, Rust 1.75+, or Java 21+

## Step 1 — Start the server (30 seconds)

Create a `docker-compose.yml`:

```yaml
services:
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"

  ojs-server:
    image: ghcr.io/openjobspec/ojs-backend-redis:latest
    ports:
      - "8080:8080"
    environment:
      REDIS_URL: redis://redis:6379
      OJS_ALLOW_INSECURE_NO_AUTH: "true"
    depends_on:
      - redis
```

Start it:

```bash
docker compose up -d
```

Verify:

```bash
curl http://localhost:8080/ojs/v1/health
# {"status":"ok","backend":{"type":"redis","status":"connected"}}
```

## Step 3 — Enqueue your first job

<Tabs>
  <TabItem label="Go">
    ```go
    // main.go
    package main

    import (
        "context"
        "fmt"
        "log"

        ojs "github.com/openjobspec/ojs-go-sdk"
    )

    func main() {
        client, err := ojs.NewClient("http://localhost:8080")
        if err != nil {
            log.Fatal(err)
        }

        job, err := client.Enqueue(context.Background(), ojs.Job{
            Type:  "greeting.say_hello",
            Queue: "default",
            Args:  []any{"world"},
        })
        if err != nil {
            log.Fatal(err)
        }

        fmt.Printf("Enqueued job %s (state: %s)\n", job.ID, job.State)
    }
    ```

    ```bash
    go run main.go
    ```

## Step 4 — Process your first job

<Tabs>
  <TabItem label="Go">
    ```go
    // worker.go
    package main

    import (
        "context"
        "fmt"
        "log"
        "os/signal"
        "syscall"

        ojs "github.com/openjobspec/ojs-go-sdk"
    )

    func main() {
        worker, err := ojs.NewWorker(ojs.WorkerConfig{
            URL:    "http://localhost:8080",
            Queues: []string{"default"},
        })
        if err != nil {
            log.Fatal(err)
        }

        worker.Register("greeting.say_hello", func(ctx context.Context, job *ojs.Job) error {
            name := job.Args[0].(string)
            fmt.Printf("Hello, %s! (job %s)\n", name, job.ID)
            return nil
        })

        ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
        defer stop()

        if err := worker.Start(ctx); err != nil {
            log.Fatal(err)
        }
    }
    ```

    ```bash
    go run worker.go
    ```
  </TabItem>
  <TabItem label="TypeScript">
    ```typescript
    // worker.mjs
    import { OJSWorker } from "@openjobspec/sdk";

    const worker = new OJSWorker({
      url: "http://localhost:8080",
      queues: ["default"],
    });

    worker.register("greeting.say_hello", async (job) => {
      const name = job.args[0];
      console.log(`Hello, ${name}! (job ${job.id})`);
    });

    await worker.start();
    console.log("Worker started. Press Ctrl+C to stop.");
    ```

    ```bash
    node worker.mjs
    ```
  </TabItem>
  <TabItem label="Python">
    ```python
    # worker.py
    from openjobspec import OJSWorker

    worker = OJSWorker(url="http://localhost:8080", queues=["default"])

    @worker.register("greeting.say_hello")
    def say_hello(job):
        name = job.args[0]
        print(f"Hello, {name}! (job {job.id})")

    if __name__ == "__main__":
        print("Worker started. Press Ctrl+C to stop.")
        worker.start()
    ```

    ```bash
    python worker.py
    ```
  </TabItem>
  <TabItem label="Ruby">
    ```ruby
    # worker.rb
    require "ojs"

    worker = OJS::Worker.new(url: "http://localhost:8080", queues: ["default"])

    worker.register("greeting.say_hello") do |job|
      name = job.args[0]

    end

    puts "Worker started. Press Ctrl+C to stop."
    worker.start
    ```

    ```bash
    ruby worker.rb
    ```
  </TabItem>
  <TabItem label="Rust">
    ```rust
    // src/worker.rs
    use ojs::{Worker, WorkerConfig, Job};

    async fn say_hello(job: &Job) -> Result<(), Box<dyn std::error::Error>> {
        let name = job.args[0].as_str().unwrap_or("stranger");
        println!("Hello, {}! (job {})", name, job.id);
        Ok(())
    }

    #[tokio::main]
    async fn main() -> Result<(), Box<dyn std::error::Error>> {
        let mut worker = Worker::new(WorkerConfig {
            url: "http://localhost:8080".into(),
            queues: vec!["default".into()],
            ..Default::default()
        })?;

        worker.register("greeting.say_hello", say_hello);

        println!("Worker started. Press Ctrl+C to stop.");
        worker.start().await
    }
    ```

    ```bash
    cargo run --bin worker
    ```
  </TabItem>
  <TabItem label="Java">
    ```java
    // WorkerDemo.java
    import org.openjobspec.sdk.OJSWorker;
    import org.openjobspec.sdk.WorkerConfig;

    public class WorkerDemo {
        public static void main(String[] args) {

                .url("http://localhost:8080")

            worker.register("greeting.say_hello", job -> {
                String name = (String) job.getArgs().get(0);

            });

            System.out.println("Worker started. Press Ctrl+C to stop.");
            worker.start();
        }
    }
    ```

    ```bash
    mvn compile exec:java -Dexec.mainClass="WorkerDemo"
    ```
  </TabItem>

## Step 5 — Monitor with the Admin UI

Start the Admin UI alongside your server:

```yaml
# Add to your docker-compose.yml
  ojs-admin-ui:
    image: ghcr.io/openjobspec/ojs-admin-ui:latest
    ports:
      - "3000:3000"
    environment:
      OJS_SERVER_URL: http://ojs-server:8080
    depends_on:
      - ojs-server
```

```bash
docker compose up -d ojs-admin-ui
```

Open [http://localhost:3000](http://localhost:3000) to see your queues, jobs, and workers in real-time.

## Step 6 — Verify via the API

You can also check job status directly via curl:

```bash
# List queues and their stats
curl http://localhost:8080/ojs/v1/queues

# Get a specific job
curl http://localhost:8080/ojs/v1/jobs/YOUR_JOB_ID

# Check the server manifest (supported extensions)
curl http://localhost:8080/ojs/manifest
```
